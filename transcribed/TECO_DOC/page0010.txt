003                     (MAIN FUNCTION IS TECO)
004                     ((TECO) (CAUSES CARRIAGE RETURN)
005                             (CAUSES BUFFER (INITIALLY NIL) TO SHOW ON THE SCREEN)
006                             (BEGINS SPECIAL READ EVAL LOOP AS FOLLOWS)
007                             (VARIOUS SINGLE LETTERS ARE READ AS)
008                             (COMMANDS
009                                     (Y (PUTS NEW SEXPR IN BUFFER)
010                                     (YOB$ GETS FIRST NON ATOM PROPERTY OF OB)
011                                     (YOB IND$ GETS THE PROPERTY IND OFF OB)
012                                     (YNONATOMSEXPR$ GETS VALUE OF SAME)
013                                     (Y SAVES OB AND IND OF BUFFER)
014                             (F (FILES CURRENT BUFFER)
015                                     (F$ FILES UNDER CURRENC IND AND OB)
016                                     (FNAME$ FILES BUFFER UNDER NAME AND CURRENT IND)
017                                     (FN1 N2$ FILES UNDER DB N1 AND IND N2))
018                             (L (MOVES POINTER LEFT OVER ONE SEXP)
019                                     (NL (MOVES LEFT OVER N SEXPRS))
020                             (R (MOVES POINTER RIGHT OVER ONE SESPR)
021                                     (NR MOVES POINTER RIGHT OVER N SEXPRS))
022                             (U (MOVES POINTER UP OUT OF CURRENT SEXPR)
023                                     (NU MOVES POINTER UP N PAREN LEVELS))
024                             (D (MOVES POINTER DOWN INTO NEXT SEXPR)
025                             (K (KILLS NEXT SEXPR)
026                                     (NK KILLS NEXT N SEXPRS))
027                             (I INSERTS
028                                     (ISTRING$ INSERTS STRING)
029                                     (STRING MAY BE AN ATO OR MORE))
030                             (S SEARCHES
031                                     (SSTRING$ SEARCHES FOR STRING)
032                                     (STRING MAY BE AN ATO OR LONGER))
033                             (E (CAUSES ENTRY TO LISTENING MOE)
034                                     ($ CAUSES RETURN TO EDIT MODE)
035                                     (P$ CAUSES TECO TO PROCEED (IE RETURN))
036                                     (EVALED EXPRS ARE DISPLAYED ON SCREEN))
037                             (B (SETS A BREAK POINT)
038                                     (B$ SETS AN UNCONDITIONAL BREAK)
039                                     (BN$ SETQ BREAK ON NTH OCCURRENCE)
040                                     (BEXPR$ SETQ BREAK IF SEXPR IS NON NIL)
041                                     ((WHEN A BREAK IS HIT)
042                                      (ONE RETURNS TO TECO IN LISTENING MODE)))
043                             (P (CAUSES TECO TO PROCEED)))) )))
044                     (PARENTHESISISED SEXPRS
045                             (ARE READ AND EVALUATED)) ))
